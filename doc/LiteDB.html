def loadpos(filepath)
  res = db.execute <<-SQL, [filepath]
    SELECT pos FROM pos_cache WHERE filepath = ?;
  SQL
  res.flatten[0] || 0
end

def savepos(filepath, pos)
  db.execute <<-SQL, [filepath, pos]
    INSERT OR REPLACE INTO pos_cache (filepath, pos) VALUES (?, ?);
  SQL
end
